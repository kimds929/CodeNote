# (Folders) 상위경로에 있는 특정 이름을 포함한 모든 폴더들을 검색
def find_folders_in_parents(folder_name, root_dir=None, partial=False, ignore_case=False, max_depth=None, get_parent_path=False):
    """
    현재 경로(root_dir)부터 상위 부모 폴더로 올라가며 조건에 맞는 폴더를 찾습니다. (BFS 방식)
    max_depth: 0이면 현재 폴더만, 1이면 부모 폴더까지, None이나 math.inf면 최상위 루트까지 탐색합니다.
    get_parent_path: True일 경우 찾은 폴더의 상위(부모) 경로를 반환합니다.
    """
    if root_dir is None:
        raise ValueError('undefined root_dir.')
    
    current = Path(root_dir).resolve()
    if current.is_file():
        current = current.parent
        
    results = []
    search_target = folder_name.lower() if ignore_case else folder_name
    limit_depth = math.inf if max_depth is None else max_depth
    
    current_depth = 0
    # 현재 폴더부터 루트까지 올라가며 확인
    for parent in [current] + list(current.parents):
        if current_depth > limit_depth:
            break
            
        try:
            for child in parent.iterdir():
                if child.is_dir():
                    compare_name = child.name.lower() if ignore_case else child.name
                    is_match = (search_target in compare_name) if partial else (search_target == compare_name)
                    
                    if is_match:
                        # get_parent_path 옵션에 따른 경로 설정
                        path_to_add = str(child.parent.absolute()) if get_parent_path else str(child.absolute())
                        # 중복 방지 로직
                        if path_to_add not in results:
                            results.append(path_to_add)
        except PermissionError:
            pass
            
        current_depth += 1
                    
    return results

# (Folders) 하위경로에 있는 특정 이름을 포함한 모든 폴더들을 검색
def find_folders_in_childs(folder_name, root_dir=None, partial=False, ignore_case=False, max_depth=None, get_parent_path=False):
    """
    root_dir 하위의 경로를 BFS(너비 우선 탐색)로 탐색하여 조건에 맞는 폴더를 찾습니다.
    max_depth: 1이면 직계 하위 폴더만, None이나 math.inf면 모든 하위 폴더를 탐색합니다.
    get_parent_path: True일 경우 찾은 폴더의 상위(부모) 경로를 반환합니다.
    """
    if root_dir is None:
        raise ValueError('undefined root_dir.')
        
    root = Path(root_dir).resolve()
    results = []
    search_target = folder_name.lower() if ignore_case else folder_name
    limit_depth = math.inf if max_depth is None else max_depth
    
    # BFS를 위한 큐 초기화: (현재_경로, 현재_깊이)
    queue = deque([(root, 0)])
    
    while queue:
        current_dir, depth = queue.popleft()
        
        # 설정한 최대 깊이에 도달하면 더 이상 하위로 내려가지 않음
        if depth >= limit_depth:
            continue
            
        try:
            for child in current_dir.iterdir():
                if child.is_dir():
                    compare_name = child.name.lower() if ignore_case else child.name
                    is_match = (search_target in compare_name) if partial else (search_target == compare_name)
                    
                    if is_match:
                        path_to_add = str(child.parent.absolute()) if get_parent_path else str(child.absolute())
                        if path_to_add not in results:
                            results.append(path_to_add)
                    
                    # 다음 깊이 탐색을 위해 큐에 추가
                    queue.append((child, depth + 1))
        except PermissionError:
            continue
                
    return results

        current = current.parent

            break
